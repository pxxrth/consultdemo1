  function initRadioPills() {
    document.querySelectorAll('.radio-pill input[type="radio"]').forEach(input => {
      input.addEventListener('change', function() {
        const name = this.name;
        // deselect all pills in this group
        document.querySelectorAll(`input[name="${name}"]`).forEach(r => {
          r.closest('.radio-pill').classList.remove('selected');
        });
        // select this one
        this.closest('.radio-pill').classList.add('selected');

        // handle conditional visibility
        const pill = this.closest('.radio-pill');
        const conditionalId = pill.dataset.shows;
        const conditionalHideId = pill.dataset.hides;
        if (conditionalId) {
          document.getElementById(conditionalId).classList.add('show');
        }
        if (conditionalHideId) {
          document.getElementById(conditionalHideId).classList.remove('show');
        }
        // also hide siblings' conditionals
        document.querySelectorAll(`input[name="${name}"]`).forEach(r => {
          const p = r.closest('.radio-pill');
          if (p !== pill && p.dataset.shows) {
            document.getElementById(p.dataset.shows).classList.remove('show');
          }
        });
      });
    });
  }
